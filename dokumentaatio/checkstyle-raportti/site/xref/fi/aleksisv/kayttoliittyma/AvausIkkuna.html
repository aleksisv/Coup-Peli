1   package fi.aleksisv.kayttoliittyma;
2   
3   import java.awt.*;
4   import javax.swing.*;
5   import javax.swing.JTextField;
6   
7   /**
8    * Luokan olio toimii pelin aloitusvalikkona.
9    */
10  public class AvausIkkuna extends JFrame {
11      
12      /** GUI, johon frame kuuluu.*/
13      private GraafinenKayttoliittyma gkl;
14      /** Nappi, jolla peli aloitetaan.*/
15      private JButton aloitaPeliNappi;
16      /** Pelaajien määrästä kertova JTextField.*/
17      private JTextField montakoPelaajaa;
18      /** Erityishuomioista kertova tekstialue.*/
19      private JTextArea huomioTekstit;
20  
21      /**
22       * Luokan konstruktori.
23       * 
24       * @param otsikko Ikkunan otsikko.
25       * @param gkl Graafinen käyttöliittymä.
26       * 
27       * @throws HeadlessException Heittää headless-exceptionin.
28       */
29      public AvausIkkuna(String otsikko, GraafinenKayttoliittyma gkl) throws HeadlessException {
30          super(otsikko);
31  
32          this.gkl = gkl;
33  
34          this.setPreferredSize(new Dimension(1000, 600));
35          this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
36  
37          luokomponentit();
38  
39          this.pack();
40          this.setVisible(true);
41  
42      }
43  
44      private void luokomponentit() {
45          Container sailio = this.getContentPane();
46          sailio.setLayout(new GridLayout(2, 2));
47          sailio.add(luoAvausvalikko());
48          this.huomioTekstit = new JTextArea("Luo peli haluamallasi määrällä pelaajia!");
49          sailio.add(huomioTekstit);
50          gkl.setHuomioTekstit(huomioTekstit);
51  
52      }
53  
54      private JPanel luoAvausvalikko() {
55          JPanel paneeli = new JPanel(new GridLayout(1, 2));
56          this.aloitaPeliNappi = new JButton("Aloita peli!");
57          this.aloitaPeliNappi.addActionListener(gkl);
58          gkl.setAloitaPeliNappi(aloitaPeliNappi);
59  
60          this.montakoPelaajaa = new JTextField();
61  
62          paneeli.add(aloitaPeliNappi);
63          paneeli.add(montakoPelaajaa);
64  
65          return paneeli;
66      }
67  
68      public JTextField getMontakoPelaajaa() {
69          return montakoPelaajaa;
70      }
71  
72      public JButton getAloitaPeliNappi() {
73          return aloitaPeliNappi;
74      }
75  
76  }
