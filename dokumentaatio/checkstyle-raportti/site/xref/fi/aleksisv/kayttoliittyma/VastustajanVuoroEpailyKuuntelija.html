1   package fi.aleksisv.kayttoliittyma;
2   
3   import fi.aleksisv.logiikka.*;
4   import java.awt.event.ActionEvent;
5   import java.awt.event.ActionListener;
6   import javax.swing.SwingUtilities;
7   
8   /**
9    * Luokka kuuntelee vastustajaan kohdistuvaa epäilyä.
10   */
11  public class VastustajanVuoroEpailyKuuntelija implements ActionListener {
12  
13      /**
14       * Onko kyseessä vuoron ensimmäinen epäilyikkuna.
15       */
16      GraafinenKayttoliittyma gkl;
17      /**
18       * Pelin pyörittämisestä vastaava taho.
19       */
20      PeliOhjaus peliOhjaus;
21      /**
22       * Pelausikkuna, joka liittyy tilanteeseen.
23       */
24      PelausIkkuna pelausIkkuna;
25      /**
26       * Tilanteeseen liittyvä vastustaja.
27       */
28      Vastustaja vastustaja;
29      /**
30       * Siirtonumero.
31       */
32      int siirto;
33  
34      /**
35       * Luokan konstruktori.
36       *
37       * @param gkl Graafinen käyttöliittymä.
38       * @param peliOhjaus Pelin pyörittämisestä vastaava taho.
39       * @param pelausIkkuna Pelaustilanteeseen liittyvä ikkuna.
40       * @param vastustaja Tilanteeseen liittyvä vastustaja.
41       * @param siirto Siirtonumero.
42       */
43      public VastustajanVuoroEpailyKuuntelija(GraafinenKayttoliittyma gkl, PeliOhjaus peliOhjaus, PelausIkkuna pelausIkkuna, Vastustaja vastustaja, int siirto) {
44          this.gkl = gkl;
45          this.peliOhjaus = peliOhjaus;
46          this.pelausIkkuna = pelausIkkuna;
47          this.vastustaja = vastustaja;
48          this.siirto = siirto;
49      }
50  
51      @Override
52      public void actionPerformed(ActionEvent e) {
53          Osanottaja o = (Osanottaja) this.vastustaja;
54          Peli peli = this.peliOhjaus.getPeli();
55          Osanottaja kohdistuu = peli.getPelaaja();
56          if (peli.getPelaaja().epaile(o, peli.getSiirtoNumerot().get(this.siirto))) {
57              this.gkl.getValiIkkuna().huomioTekstit.setText("Onnistuit epäilemään vastustajan siirtoa.");
58              peli.setVuoronumero(peli.getVuoronumero() + 1);
59              this.peliOhjaus.paivitaTilanne();
60          } else {
61              this.peliOhjaus.suoritaSiirto(o, kohdistuu, siirto);
62              this.gkl.getValiIkkuna().huomioTekstit.setText("Osanottaja " + o.getNimi()
63                      + " kohdisti  siirron " + peli.getSiirtoNimet().get(siirto) + " osanottajaa "
64                      + kohdistuu.getNimi() + " kohtaan.");
65          }
66  
67          this.gkl.paivitaPelinseuranta(this.gkl.getPelinSeurantapaneeli());
68          this.pelausIkkuna.setVisible(false);
69          SwingUtilities.updateComponentTreeUI(this.gkl.getValiIkkuna());
70      }
71  
72  }
