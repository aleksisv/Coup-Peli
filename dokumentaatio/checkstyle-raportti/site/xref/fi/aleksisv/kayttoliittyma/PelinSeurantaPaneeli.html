6   package fi.aleksisv.kayttoliittyma;
7   
8   import fi.aleksisv.logiikka.Osanottaja;
9   import fi.aleksisv.logiikka.Pelaaja;
10  import fi.aleksisv.logiikka.Peli;
11  import java.awt.Dimension;
12  import java.awt.GridLayout;
13  import java.awt.Insets;
14  import java.util.ArrayList;
15  import javax.swing.JPanel;
16  import javax.swing.JTextArea;
17  import javax.swing.SwingUtilities;
18  
19  public class PelinSeurantaPaneeli extends JPanel {
20      
21      /**
22       * Luokan konstruktori.
23       * 
24       * @param layout Mikä layout paneelille halutaan.
25       */
26      public PelinSeurantaPaneeli(GridLayout layout) {
27          super(layout);
28          this.setVisible(true);
29      }
30  
31      /**
32       * Metodi asettaa pelin tilanne-esityksen alkutilaansa.
33       *
34       * @param peli Peli, jota tällä hetkellä käsitellään.
35       */
36      public void asetaAlkutila(Peli peli) {
37          for (Osanottaja o : peli.getOsanottajajoukko()) {
38              if (o instanceof Pelaaja) {
39                  JTextArea teksti = new JTextArea("Pelaaja: \n" + o.toString());
40                  teksti.setMargin(new Insets(5, 5, 5, 5));
41                  teksti.setSize(new Dimension(100, 100));
42                  super.add(teksti);
43              } else {
44                  JTextArea teksti = new JTextArea("Vastustaja: \n" + o.toString());
45                  teksti.setSize(new Dimension(100, 100));
46                  teksti.setMargin(new Insets(5, 5, 5, 5));
47                  super.add(teksti);
48              }
49          }
50          SwingUtilities.updateComponentTreeUI(this);
51      }
52  
53      /**
54       * Metodin avulla pelinseurannan tila päivitetään.
55       *
56       * @param peli Mikä peli päivitetään.
57       */
58      public void paivitaTila(Peli peli) {
59          ArrayList<Osanottaja> kaikki = (ArrayList<Osanottaja>) peli.getOsanottajajoukko().clone();
60          kaikki.addAll(peli.getHavinnytJoukko());
61          for (Osanottaja o : kaikki) {
62              if (o instanceof Pelaaja) {
63                  JTextArea teksti = new JTextArea("Pelaaja: \n" + o.toString());
64                  teksti.setSize(new Dimension(80, 80));
65                  teksti.setMargin(new Insets(10, 10, 10, 10));
66                  super.add(teksti);
67              } else {
68                  JTextArea teksti = new JTextArea("Vastustaja: \n" + o.toString());
69                  teksti.setSize(new Dimension(80, 80));
70                  teksti.setMargin(new Insets(10, 10, 10, 10));
71                  super.add(teksti);
72              }
73          }
74          SwingUtilities.updateComponentTreeUI(this);
75      }
76  
77  }
