1   package fi.aleksisv.kayttoliittyma;
2   
3   import fi.aleksisv.logiikka.Vastustaja;
4   import java.awt.Container;
5   import java.awt.GridLayout;
6   import javax.swing.JButton;
7   import javax.swing.JFrame;
8   import javax.swing.JPanel;
9   import javax.swing.JTextArea;
10  
11  /**
12   * Luokka kuvaa pelaajan torjuntatilanteeseen liittyvää ikkunaa.
13   */
14  public class Torjuntaikkuna extends JFrame {
15  
16      /**
17       * Vastustaja, joka liittyy tilanteeseen.
18       */
19      Vastustaja vastustaja;
20      /**
21       * Pelin pyörittämisestä vastaava taho.
22       */
23      PeliOhjaus peliOhjaus;
24      /**
25       * Siirtonumero.
26       */
27      int siirto;
28      /**
29       * Graafinen käyttöliittymä.
30       */
31      GraafinenKayttoliittyma gkl;
32  
33      /**
34       * Luokan konstruktori.
35       * @param vastustaja Tilanteeseen liittyvä vastustaja.
36       * @param peliOhjaus Pelin pyörittämisestä vastaava taho.
37       * @param siirto Siirtonumero
38       * @param gkl Graafinen käyttöliittymä.
39       */
40      public Torjuntaikkuna(Vastustaja vastustaja, PeliOhjaus peliOhjaus, int siirto, GraafinenKayttoliittyma gkl) {
41          super("Torjuntaikkuna");
42          this.vastustaja = vastustaja;
43          this.peliOhjaus = peliOhjaus;
44          this.siirto = siirto;
45          this.gkl = gkl;
46      }
47      
48      /** 
49       * Metodi luo tarvittavat komponentit.
50       * @param sailio Container-olio.
51       */
52      public void luoKomponentit(Container sailio) {
53          JPanel torjuntaPaneeli = new JPanel(new GridLayout(3, 1));
54          sailio.setVisible(true);
55  
56          JButton epaileVastustajanTorjuntaa = new JButton("Epäile vastustajan torjuntaa.");
57          JButton peruutaSiirto = new JButton("Peruuta siirto.");
58  
59          Epailykuuntelija epailykuuntelija = new Epailykuuntelija(epaileVastustajanTorjuntaa,
60                  peruutaSiirto, false, vastustaja, peliOhjaus, siirto, gkl, this);
61  
62          epaileVastustajanTorjuntaa.addActionListener(epailykuuntelija);
63          peruutaSiirto.addActionListener(epailykuuntelija);
64          torjuntaPaneeli.add(new JTextArea("Vastustaja haluaa torjua siirtosi. Haluatko epäillä vastustajan torjumista?"));
65          torjuntaPaneeli.add(epaileVastustajanTorjuntaa);
66          torjuntaPaneeli.add(peruutaSiirto);
67  
68          sailio.add(torjuntaPaneeli);
69          sailio.setSize(700, 400);
70  
71          this.setVisible(true);
72  
73          this.pack();
74      }
75  
76  }
