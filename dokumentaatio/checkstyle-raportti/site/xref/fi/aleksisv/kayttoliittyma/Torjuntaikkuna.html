3   import fi.aleksisv.logiikka.Vastustaja;
4   import java.awt.BorderLayout;
5   import java.awt.Color;
6   import java.awt.Container;
7   import java.awt.GridLayout;
8   import javax.swing.BorderFactory;
9   import javax.swing.JButton;
10  import javax.swing.JFrame;
11  import javax.swing.JPanel;
12  import javax.swing.JTextArea;
13  
14  /**
15   * Luokka kuvaa pelaajan torjuntatilanteeseen liittyvää ikkunaa.
16   */
17  public class Torjuntaikkuna extends JFrame {
18  
19      /**
20       * Vastustaja, joka liittyy tilanteeseen.
21       */
22      Vastustaja vastustaja;
23      /**
24       * Pelin pyörittämisestä vastaava taho.
25       */
26      PeliOhjaus peliOhjaus;
27      /**
28       * Siirtonumero.
29       */
30      int siirto;
31      /**
32       * Graafinen käyttöliittymä.
33       */
34      GraafinenKayttoliittyma gkl;
35  
36      /**
37       * Luokan konstruktori.
38       * @param vastustaja Tilanteeseen liittyvä vastustaja.
39       * @param peliOhjaus Pelin pyörittämisestä vastaava taho.
40       * @param siirto Siirtonumero
41       * @param gkl Graafinen käyttöliittymä.
42       */
43      public Torjuntaikkuna(Vastustaja vastustaja, PeliOhjaus peliOhjaus, int siirto, GraafinenKayttoliittyma gkl) {
44          super("Torjuntaikkuna");
45          this.vastustaja = vastustaja;
46          this.peliOhjaus = peliOhjaus;
47          this.siirto = siirto;
48          this.gkl = gkl;
49          this.setLocation(0, 500);
50      }
51      
52      /** 
53       * Metodi luo tarvittavat komponentit.
54       * @param sailio Container-olio.
55       */
56      public void luoKomponentit(Container sailio) {
57          JPanel torjuntaPaneeli = new JPanel(new GridLayout(3, 1));
58          torjuntaPaneeli.setBounds(30, 30, 30, 30);
59          
60          sailio.setVisible(true);
61          
62          JTextArea teksti = new JTextArea("Vastustaja haluaa torjua siirtosi.\nHaluatko epäillä vastustajan torjumista?");
63          
64          JPanel ilmoitusPaneeli = new JPanel(new BorderLayout());
65          ilmoitusPaneeli.setBorder(BorderFactory.createEtchedBorder(Color.lightGray, Color.black));
66          
67         
68          JButton epaileVastustajanTorjuntaa = new JButton("Epäile vastustajan torjuntaa.");
69          JButton peruutaSiirto = new JButton("Peruuta siirto.");
70  
71          Epailykuuntelija epailykuuntelija = new Epailykuuntelija(epaileVastustajanTorjuntaa,
72                  peruutaSiirto, false, vastustaja, peliOhjaus, siirto, gkl, this);
73          epaileVastustajanTorjuntaa.addActionListener(epailykuuntelija);
74          
75          peruutaSiirto.addActionListener(epailykuuntelija);
76          
77          ilmoitusPaneeli.add(teksti, BorderLayout.CENTER);
78          
79          torjuntaPaneeli.add(ilmoitusPaneeli);
80          
81          torjuntaPaneeli.add(epaileVastustajanTorjuntaa);
82          torjuntaPaneeli.add(peruutaSiirto);
83  
84          sailio.add(torjuntaPaneeli);
85          
86          this.setVisible(true);
87          this.setSize(400, 500);
88          
89      }
90  
91  }
