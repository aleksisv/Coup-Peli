3   import java.awt.*;
4   import java.awt.event.WindowEvent;
5   import javax.swing.*;
6   import javax.swing.JTextField;

8   /**
9    * Luokan olio toimii pelin aloitusvalikkona.
10   */
11  public class AvausIkkuna extends JFrame {
12  
13      /**
14       * GUI, johon frame kuuluu.
15       */
16      private GraafinenKayttoliittyma gkl;
17      /**
18       * Nappi, jolla peli aloitetaan.
19       */
20      private JButton aloitaPeliNappi;
21      /**
22       * Pelaajien määrästä kertova JTextField.
23       */
24      private JTextField montakoPelaajaa;
25      /**
26       * Pelaajan nimestä kertova JTextField.
27       */
28      private JTextField pelaajanNimi;
29      /**
30       * Erityishuomioista kertova tekstialue.
31       */
32      private JTextArea huomioTekstit;
33  
34      /**
35       * Luokan konstruktori.
36       *
37       * @param otsikko Ikkunan otsikko.
38       * @param gkl Graafinen käyttöliittymä.
39       *
40       * @throws HeadlessException Heittää headless-exceptionin.
41       */
42      public AvausIkkuna(String otsikko, GraafinenKayttoliittyma gkl) throws HeadlessException {
43          super(otsikko);
44  
45          this.gkl = gkl;
46  
47          this.setPreferredSize(new Dimension(1000, 600));
48          this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
49  
50          luokomponentit(this.getContentPane());
51  
52          this.pack();
53          this.setVisible(true);
54  
55      }
56  
57      private void luokomponentit(Container sailio) {
58          sailio.setLayout(new GridLayout(2, 1));
59          sailio.add(luoAvausvalikko());
60          this.huomioTekstit = new JTextArea("Tämä on Coup-peli. Tehtävänäsi on paljastaa vastustajiesi kortit ja suojella omiasi. \n"
61                  + "Saat pelin alussa kaksi korttia. Jokaisella kortilla on oma. Joka vuoro joudut tekemään siirron, joista johoinkin vaaditaan"
62                  + " tietyn kortin omistamista.\nVastustaja voi yrittää torjua siirron tai epäillä, että sinulla ei ole korttia."
63                  + "Voit tehdä samoin. Jos epäilijä on oikeassa eikä siirron tekijällä ole korttia, siirron tekijä menettää yhden kortin.\n"
64                  + "Vastaavasti jos epäilijä on väärässä epäilijä itse menettää kortin ja siirto osuu häneen."
65                  + "\n\nValitse sopiva nimi itsellesi ja luo peli haluamallasi määrällä pelaajia!"
66                  + "\nSallittu määrä pelaajia on 2-5.");
67          this.huomioTekstit.setBorder(BorderFactory.createEtchedBorder(Color.lightGray, Color.black));
68          JPanel tekstiPaneeli = new JPanel(new BorderLayout());
69          tekstiPaneeli.setBorder(BorderFactory.createEmptyBorder(20, 20, 20, 20));
70          tekstiPaneeli.add(huomioTekstit);
71          sailio.add(tekstiPaneeli);
72          gkl.setHuomioTekstit(huomioTekstit);
73  
74      }
75  
76      private JPanel luoAvausvalikko() {
77          JPanel paneeli = new JPanel(new GridLayout(1, 3));
78  
79          JPanel aloitusSala = new JPanel(new BorderLayout());
80          this.aloitaPeliNappi = new JButton("Aloita peli!");
81          this.aloitaPeliNappi.setSize(40, 40);
82  
83          aloitusSala.setBorder(BorderFactory.createEmptyBorder(135, 20, 20, 20));
84          aloitusSala.add(this.aloitaPeliNappi, BorderLayout.CENTER);
85          this.aloitaPeliNappi.addActionListener(gkl);
86          gkl.setAloitaPeliNappi(aloitaPeliNappi);
87  
88          JPanel osanottajia = new JPanel(new GridLayout(2, 1));
89          osanottajia.setBorder(BorderFactory.createEmptyBorder(100, 20, 20, 20));
90          osanottajia.add(new JLabel("Monellako pelaajalla haluat pelata?"));
91          this.montakoPelaajaa = new JTextField();
92          osanottajia.add(montakoPelaajaa);
93  
94          JPanel nimi = new JPanel(new GridLayout(2, 1));
95          nimi.setBorder(BorderFactory.createEmptyBorder(100, 20, 20, 20));
96          nimi.add(new JLabel("Minkä nimen haluat?"));
97          this.pelaajanNimi = new JTextField();
98          nimi.add(pelaajanNimi);
99  
100         paneeli.add(aloitusSala);
101         paneeli.add(osanottajia);
102         paneeli.add(nimi);
103 
104         return paneeli;
105     }
106

110

112         return aloitaPeliNappi;
113     }
114 
115     public JTextField getPelaajanNimi() {
116         return pelaajanNimi;
117     }
