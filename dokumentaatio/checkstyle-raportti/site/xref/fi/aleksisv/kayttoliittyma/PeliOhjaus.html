29          int i = Integer.parseInt(lukija.nextLine());

34      private void luoPeli(int pelaajaMaara) {
35          this.peli = new Peli(pelaajaMaara);
36          peli.luoOsanottajat();
37      }

39      private void pelaa() {
40          while (this.peli.getOsanottajajoukko().size() > 1) {
41              siirtovalinta(this.peli.getOsanottajajoukko().get(this.peli.getVuoronumero() % this.peli.getOsanottajajoukko().size()));
42              ilmoitaSiirtovalinnasta(this.peli.getOsanottajajoukko().get(this.peli.getVuoronumero() % this.peli.getOsanottajajoukko().size()), 1);
43              this.peli.setVuoronumero(this.peli.getVuoronumero() + 1);
44          }
45      }

47      private void siirtovalinta(Osanottaja o) {
48          if (o instanceof Pelaaja) {
49              pelaajanSiirtovalinta((Pelaaja) o);
50          } else if (o instanceof Vastustaja) {
51              vastustajanSiirtovalinta((Vastustaja) o);
52          }
53      }

55      private void pelaajanSiirtovalinta(Pelaaja pelaaja) {
56          System.out.println("Vuoro numero " + this.peli.getVuoronumero() + ". \n");
57          System.out.println("Minkä siirron haluat tehdä?");
58          System.out.println("Vaihtoehdot: (1) Income (ota 1 kolikko pankista)\n"
59                  + "             (2) Foreign Aid (ota kaksi kolikkoa pankista)\n"
60                  + "             (3) Coup (maksaa seitsemän kolikkoa, tapa mikä tahansa kortti)\n"
61                  + "             (4) Taxes (ota kolme kolikkoa pankista)\n"
62                  + "             (5) Assassinate (maksa kolme kolikkoa, tapa mikä tahansa kortti)\n"
63                  + "             (6) Steal (ota kaksi kolikkoa toiselta pelaajalta)\n"
64                  + "             (7) Swap Influence (nosta kaksi korttia pakasta ja vaihda yksi omaan pakkaasi).");
65  
66          while (true) {
67              int i = Integer.parseInt(lukija.nextLine());
68              if (1 <= i && i <= 7) {
69                  System.out.println("Valitsit vaihtoehdon " + i + ".\n");
70                  if (i == 1) {
71                      pelaaja.kaytaBasicIncome(this.peli.getPankki());
72                  } else if (i == 2) {
73                      pelaaja.kaytaForeignAid(this.peli.getPankki());
74                  } else if (i == 3) {
75                      if (pelaaja.getRaha() < 7) {
76                          System.out.println("Sinulla ei ole rahaa tähän siirtoon. Valitse joku toinen. \n");
77                      }
78                      System.out.println("Kenet haluat Coupata?");
79                      int coupattava = Integer.parseInt(lukija.nextLine());
80                      pelaaja.kaytaCoup(this.peli.getPankki(), this.peli.getOsanottajajoukko().get(coupattava));
81                  } else if (i == 4) {
82                      pelaaja.kaytaTaxes(this.peli.getPankki());
83                  } else if (i == 5) {
84                      if (pelaaja.getRaha() < 3) {
85                          System.out.println("Sinulla ei ole rahaa tähän siirtoon. Valitse joku toinen. \n");
86                      }
87                      System.out.println("Kenet haluat Assassinoida");
88                      int assassinoitava = Integer.parseInt(lukija.nextLine());
89                      pelaaja.kaytaAssassinate(this.peli.getPankki(), this.peli.getOsanottajajoukko().get(assassinoitava));
90                  } else if (i == 6) {
91                      System.out.println("Keneltä haluat varastaa?");
92                      int varkaudenKohde = Integer.parseInt(lukija.nextLine());
93                      pelaaja.kaytaSteal(this.peli.getPankki(), this.peli.getOsanottajajoukko().get(varkaudenKohde));
94                  } else if (i == 7) {
95                      System.out.println("Toiminnallisuutta ei vielä tueta.");
96                  }
97              }
98              System.out.println("Anna sääntöjen mukainen vastaus. \n");
99          }
100     }

104         return r.nextInt(7) + 1;

107     private void ilmoitaSiirtovalinnasta(Osanottaja osanottaja, int siirtoVaihtoehto) {
108         System.out.println("Osanottaja " + osanottaja + " haluaa yrittää siirtoa " + siirtoVaihtoehto + ".");
109         boolean pelaajanEpaily = false;
110         boolean pelaajanBlokkaus = true;
111         Vastustaja vastustajaJokaEpailee = this.peli.kukaVastustajiastaEpailee(osanottaja, siirtoVaihtoehto);
112         boolean vastustajanBlokki = this.peli.haluaakoJokuVastustajiastaBlokata(osanottaja, siirtoVaihtoehto);
113         if (!osanottaja.equals(this.peli.getPelaaja())) {
114             System.out.println("\nHaluatko epäillä siirtoa?");
115             pelaajanEpaily = Boolean.parseBoolean(lukija.nextLine());
116             System.out.println("\nHaluatko blokata siirron?");
117             pelaajanBlokkaus = Boolean.parseBoolean(lukija.nextLine());
118         }
119 
120         if (vastustajanBlokki) {
121             return;
122         } else if (pelaajanBlokkaus) {
123             return;
124         } else if (pelaajanEpaily) {
125             this.peli.getPelaaja().epaileSiirtoa(osanottaja, siirtoVaihtoehto);
126         } else if (!(vastustajaJokaEpailee == null)) {
127             vastustajaJokaEpailee.epaileSiirtoa(osanottaja, siirtoVaihtoehto);
128         } else {
129             suoritaSiirto(osanottaja, siirtoVaihtoehto);
130         }
131     }
